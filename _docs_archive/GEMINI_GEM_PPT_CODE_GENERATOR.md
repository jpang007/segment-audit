```python
def add_limits_slide(prs):
```

```python
def add_events_slide(prs):
    """Top events with horizontal bar chart"""
    slide = prs.slides.add_slide(prs.slide_layouts[6])
    add_slide_title(slide, "Data: Events & Props")
    
    # Horizontal bar chart - top events
    chart_data = CategoryChartData()
    chart_data.categories = [e["event"] for e in TOP_EVENTS[:10]]
    chart_data.add_series('Volume', [e["volume"] for e in TOP_EVENTS[:10]])
    
    x, y, cx, cy = Inches(0.5), Inches(1.5), Inches(5.5), Inches(3.5)
    chart = slide.shapes.add_chart(
        XL_CHART_TYPE.BAR_CLUSTERED, x, y, cx, cy, chart_data
    ).chart
    
    chart.has_legend = False
    chart.has_title = True
    chart.chart_title.text_frame.text = "Top Events by Volume"
    
    # Color bars to match template
    series = chart.series[0]
    for idx, point in enumerate(series.points):
        point.format.fill.solid()
        # Alternate colors like template
        color = Colors.BLUE if idx % 2 == 0 else Colors.TEAL
        point.format.fill.fore_color.rgb = color
```
